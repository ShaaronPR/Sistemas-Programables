```c++
//Variables
const int touchPin = 4;
const int ledPin = 16;

//Numero m谩ximo que retorna el ESP32 cuando se toca el sensor
//Punto entre los valores de cuando se toca el sensor y cuando no.
const int limite = 20;
int touchValor;
const int freq = 5000; //Hz - Frecuencia programable hasta 240MHz
const int ledCanal = 0;
const int resolucion = 8;

void setup()
{
    //Velocidad de datos en bits por segundo
    Serial.begin(9600);
    delay(1000);
    
    //Pin donde est谩 conectado el led como (pin de salida).
    pinMode(ledPin, OUTPUT);
    
    //Funci贸n que inicializa la configuraci贸n del PWM (Modulaci贸n por ancho de pulsos), 
    //Par谩metros: canal Led, frecuencia y la resoluci贸n.
    ledcSetup(ledCanal, freq, resolucion);
    
    //Se relaciona el pin GPIO (General Purpose Input Output) sistema de entrada y salida 
    //de prop贸sito general con el canal del PWM.
    ledcAttachPin(ledPin, ledCanal);  //Adjunta el pin al canal PWM
}


void loop()
{
    //Leer el sensor t谩ctil es sencillo. En IDE de Arduino, usa el touchRead() como una funci贸n, 
    //que acepta como argumento, el touchPin que desea leer.
    touchValor = touchRead(touchPin);
    
    //Si este valor medido es menor al l铆mite establecido enciende el LED.
    if (touchValor < limite)
    {
        //Si el estado del led es HIGH (encendido), entra a la primera condici贸n
        if (digitalRead(ledPin))
        {
            //Se enciende y se apague el LED en un tiempo de 3s.
            ledcWrite(ledCanal, 0);
            delay(750);
            ledcWrite(ledCanal, 255);
            delay(750);
            ledcWrite(ledCanal, 0);
            delay(750);
            ledcWrite(ledCanal, 255);
            delay(750);
            ledcWrite(ledCanal, 0);
        }

        //Si el estado del led es LOW (apagado)...
        else
        {
            //Ciclo para subir la intensidad del LED de 0 a 255 porque se eligi贸 una resoluci贸n de 
            //8 bits = 256
            for (int cicloTrabajo = 0; cicloTrabajo <= 255; cicloTrabajo++)
            {
                ledcWrite(ledCanal, cicloTrabajo);
                delay(15);
            }
        }
    }
}
```
